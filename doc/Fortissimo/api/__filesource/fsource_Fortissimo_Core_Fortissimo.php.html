  2. /**
  3.  * The Fortissimo core.
  4.  *
  5.  * This file contains the core classes necessary to bootstrap and run an
  6.  * application that makes use of the Fortissimo framework. All of the necessary
  7.  * classes are encapsulated in this single file so that bootstrapping time can be
  8.  * kept to a minimum, with as little loading/autoloading overhead as possible.
  9.  *
  10.  * When this file is loaded, the include path is augmented to include the
  11.  * includes/, core/, and phar/ directories.
  12.  *
  13.  * <b>Using Fortissimo</b>
  14.  * Fortissimo is a framework for developing web applications. Unlike most
  15.  * web frameworks, it is not based on the MVC pattern. Instead, it uses
  16.  * the front controller pattern, coupled with a variation on the chain-of-command
  17.  * pattern.
  18.  *
  19.  * To get started, look at the command.xml file. The front controller reads
  20.  * this file to map requests to a sequence of commands. When building a
  21.  * Fortissimo application, you will be building such mappings yourself.
  22.  *
  23.  * Each request will kick off a chain of zero or more commands, which are
  24.  * executed in sequence. A command is simply a class that implements the
  25.  * {@link FortissimoCommand} class. Most commands will extend the abstract
  26.  * {@link BaseFortissimoCommand} class, which provides a baseline of
  27.  * functionality.
  28.  *
  29.  * A command can take zero or more parameters. Parameters can be retrieved from
  30.  * a wide variety of sources, the most common being GET and POST data, and
  31.  * output from other commands. Since commands are highly configurable, they
  32.  * are input-neutral: the parameters can come from any input. Fortissimo
  33.  * itself handles the retrieval of parameters, and the source of each parameter
  34.  * is configured not in code, but in the commands.xml file. Practically speaking,
  35.  * what this means is that one command can recieve input from a user-configurable
  36.  * source. The data can come from GET, POST, cookies, sessions, other commands,
  37.  * command line args, or environment variables.
  38.  *
  39.  * To get started with Fortissimo, take a look at some of the unit testing
  40.  * classes or read the online documentation. Then write a custom class implementing
  41.  * the BaseFortissimoCommand.
  42.  *
  43.  * <b>The Fortissimo Autoloader</b>
  44.  * Fortissimo uses the default SPL autoloader to find and load classes. For that
  45.  * reason, if you follow the convention of storing classes inside of files
  46.  * named with the class name, you will not need to use {@link include()} or any
  47.  * other such functions.
  48.  *
  49.  * Current benchmarks indicate that the SPL autoloader is slightly faster than
  50.  * using require/require_once, both with and without an opcode cache.
  51.  *
  52.  * You can augment the autoloader's default paths using <code>include</code>
  53.  * elements in the command.xml file (or manually loading them in via the
  54.  * config object).

  150.    * scale linearly (dependent, of course, on the commands themselves). However,
  151.    * since the typical PHP application handles only one request per invocation,
  152.    * this controller will attempt to bootstrap very quickly with minimal loading.
  153.    *
  154.    * It should be illegal to eat bananas on a crowded train. They smell bad, and

  160.    *   and a SimpleXML object are among the various objects that can be passed

  229.     if (empty($request)) {
  230.       throw new FortissimoException('Request not found.');
  231.     }
  232.

  237.         $out .= $command['instance']->explain();
  238.       }
  239.       else {

  241.         
  242.         $out .= sprintf($filter, $command['name'], $command['class']) . PHP_EOL;
  243.       }
  244.     }
  245.     return $out . PHP_EOL;

  395.     }
  396.     catch (Exception $fatal) {
  397.       restore_error_handler();

  399.     }

  411.    */
  412.   protected function fetchParameters($commandArray) {
  413.     $params = array();
  414.     foreach ($commandArray['params'] as $name => $config) {
  415.       
  416.       
  417.       // If there is a FROM source, fetch the data from the designated source(s).
  418.       if (!empty($config['from'])) {
  419.         // Handle cases like this: 'from="get:preferMe post:onlyIfNotInGet"'
  420.         $fromItems = explode(' ', $config['from']);
  421.         $value = NULL;
  422.         
  423.         // Stop as soon as a parameter is fetched and is not NULL.
  424.         foreach ($fromItems as $from) {
  425.           $value = $this->fetchParameterFromSource($from);
  426.           if (isset($value)) {
  427.             $params[$name] = $value;
  428.             break;
  429.           }
  430.         }
  431.       }
  432.             
  433.       // Set the default value if necessary.
  434.       if (!isset($params[$name])) $params[$name] = $config['value'];
  435.     }
  436.     return $params;
  437.   }
  438.   
  439.   /**
  440.    * Parse a parameter specification and retrieve the appropriate data.
  441.    *
  442.    * @param string $from 
  443.    *   A parameter specification of the form <source>:<name>. Examples:
  444.    *   - get:myParam
  445.    *   - post:username
  446.    *   - cookie:session_id
  447.    *   - session:last_page
  448.    *   - cmd:lastCmd
  449.    *   - env:cwd
  450.    *   Known sources:
  451.    *   - get
  452.    *   - post
  453.    *   - cookie
  454.    *   - session
  455.    *   - cmd (retrieved from the execution context.)
  456.    *   - env
  457.    *   - server
  458.    *   - request
  459.    *   - argv (From $argv, assumes that the format of from is argv:N, where N is an integer)
  460.    * @param FortissimoExecutionContext $cxt 
  461.    *   The current working context. This is used to retrieve data from cmd:
  462.    *   sources.
  463.    * @return string 
  464.    *   The value or NULL.
  465.    */
  466.   protected function fetchParameterFromSource($from) {
  467.     list($proto, $paramName) = explode(':', $from, 2);
  468.     $proto = strtolower($proto);
  469.     switch ($proto) {
  470.       case 'g':
  471.       case 'get':
  472.         return $_GET[$paramName];
  473.       case 'p':
  474.       case 'post':
  475.         return $_POST[$paramName];
  476.       case 'c':
  477.       case 'cookie':
  478.       case 'cookies':
  479.         return $_COOKIE[$paramName];
  480.       case 's':
  481.       case 'session':
  482.         return $_SESSION[$paramName];
  483.       case 'x':
  484.       case 'cmd':
  485.       case 'cxt':
  486.       case 'context':
  487.         return $this->cxt->get($paramName);
  488.       case 'e':
  489.       case 'env':
  490.       case 'environment':
  491.         return $_ENV[$paramName];
  492.       case 'server':
  493.         return $_SERVER[$paramName];
  494.       case 'r':
  495.       case 'request':
  496.         return $_REQUEST[$paramName];
  497.       case 'a':
  498.       case 'arg':
  499.       case 'argv':
  500.         return $argv[(int)$paramName];
  501.     }
  502.   }
  503. }
